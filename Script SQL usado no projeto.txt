create database cadastro;

use cadastro;

create table cliente(
	Id INT PRIMARY KEY AUTO_INCREMENT,
	Nome VARCHAR(30) NOT NULL,
	Cpfcnpj VARCHAR(18) UNIQUE,
	Endereco VARCHAR(100)
	) ENGINE=InnoDB;

create table item(
	Id INT PRIMARY KEY AUTO_INCREMENT,
	Descricao VARCHAR(40) NOT NULL,
	Preco FLOAT(8,2) UNIQUE
	) ENGINE=InnoDB;
